面向对象的克隆与继承

继承

比如两个构造函数,构造函数二(子函数)要继承构造函数一(父函数):

1–直接在要继承的函数内(子函数)内用apply或call指向富函数
//父函数
function Preson(name,age){
    this.name=name;
    this.age=age;
    this.say=function(){
    alert('我叫'+this.name)
    }
}
//子函数
function Girl(name,age){

    Preson.apply(this,arguments)//把Preson所有的属性和方法继承到Girl里(只能继承Preson构造函数内的属性和方法,Preson的prototype上的方法和属性继承不到)
    //或者Preson.call(this,[name,age])
    this.shopping=function(){//自己的方法
        alert('买东西')
    }
}

var xiaoming =new Preson('小明',18)
var xiaohong =new Girl('小红',16)

//父函数
function Preson(name,age){
    this.name=name;
    this.age=age;
    this.say=function(){
    alert('我叫'+this.name)
    }
}

Preson.prototype.dance=function(){
    alert('跳舞')
}//如果在Preson原型上加上方法,那么子函数就继承不到


//子函数
function Girl(name,age){

    Preson.apply(this,arguments)//把Preson所有的属性和方法继承到Girl里(只能继承Preson构造函数内的属性和方法,Preson的prototype上的方法和属性继承不到)
    //或者Preson.call(this,[name,age])
    this.shopping=function(){//自己的方法
        alert('买东西')
    }
}

var xiaoming =new Preson('小明',18)
var xiaohong =new Girl('小红',16)

xiaohong.dance()
//显示结果Cannot read property 'dance' of undefined无法定义
由此可call和apply的方式继承不到符的构造函数的原型对象上的属性和方法

通过父构造函数的实力赋予子构造函数的原型来实现继承,父构造函数中的属性没办法传参

//父函数
function Preson(name,age){
    this.name=name;
    this.age=age;
    this.say=function(){
    alert('我叫'+this.name)
    }
}

Preson.prototype.dance=function(){
    alert('跳舞')

    Preson.apply(this,arguments)//把Preson所有的属性和方法继承到Girl里(只能继承Preson构造函数内的属性和方法,Preson的prototype上的方法和属性继承不到)
    //或者Preson.call(this,[name,age])

function Preson(name,age){
    this.name=name;
    this.age=age;
    this.say=function(){
    alert('我叫'+this.name)
    }
}

Preson.prototype.dance=function(){
    alert('跳舞')
}


//子函数
function Girl(name,age){

    Preson.apply(this,arguments)        //或者Preson.call(this,[name,age])
    this.shopping=function(){//自己的方法
        alert('买东西')
    }
}

function f(){//通过一个自定义构造函数传递
}
f.prototype=Preson.prototype

Girl.prototype=new f()//自定义函数实例化Girl的原型
var xiaoming =new Preson('小明',18)
var xiaohong =new Girl('小红',16)

xiaohong.dance()//可以弹出'跳舞'

克隆

在Javascript中又这样一个现象

var a=1;//基本数据类型
var b=a;
b=2;

alert(a,b)//弹出窗口(1,2)


var a=[1,2,3,4]//复杂数据类型
var b=a;
b[0]=10;

alert(a,b)//弹出窗口[10,2,3,4],[10.2,3,4]

这种情况中基本数据类型可以改变值而不影响a值,是复制过来这就是传值,而复杂数据类型随着第二个变量的改变第一个也会改变,这就是传址

传值:把一个变量的值复制一份给另一个变量.

1)复制变量的值

2)申请一个新的内存地址存复制的值

3)把复制的值的地址告诉另一个变量

传址:把一个变量在内存中的地址复制给另一个变量

复杂的数据类型的赋值是传址

方法一不改变对象一的值

var a=[1,2,3,4];
var b=[];

for(var i=0;i<a.length;i++){
    b[i]=a[i]
}//通过for循环来给变量b赋值

b[1]=10;

alert(a,b)//弹出窗口[1,2,3,4],[10.2,3,4]

方法二

var a=[1,2,3,4];
var b=[];

a.forEach(function(val,index,arr){
    b[index]=a[index]
},1)//通过forEach给变量b赋值,forEach和for循环式有区别的,他能够循环更为复杂的数据类型比如[a,b,[a,s],{f:123,t:'dsad'}]


b[1]=10;

alert(a,b)//弹出窗口[1,2,3,4],[10.2,3,4]

方法三

先了解一下JSON的两个方法
JSON.stringify()—-把对象转换成字符串形式的JSON格式字符串值,且只能转换js对象,不能转换Dom和Bom对象

JSON.parse()
把字符串形式的标准json字符串转换成对象.(注意是JSON字符串必须是标准格式,属性名称必须有引号)

var arr1=[1,2,[5,22],{a:'ddd',b:true}];

var arr2=JSON.stringify(arr1)//把对象arr1转换为json字符串赋给arr2

arr2=JSON.parse(arr2);//把自身的JSON格式的字符串转换为对象

arr2[2][0]=10;

console.log(arr1,arr2)//可以看出[1,2,[5,22],{a:'ddd',b:true}],[1,2,[10,22],{a:'ddd',b:true}]
